Sass - Nesting


/* Sass */
nav {
	ul {
		list-style: none;
		margin: 0;
		padding: 0;
		overflow: hidden; /* clearfixer */
	}
	li {
		float: left;
	}
	a {
		display: block;
	}
}
					

/* output */
nav ul {
	list-style: none;
	margin: 0;
	padding: 0;
	overflow: hidden; /* clearfixer */
}
nav li {
	float: left;
}
nav a {
	display: block;
}

Sass - Mixins


@mixin border-radius($radius) {
	-webkit-border-radius: $radius;
	-moz-border-radius: $radius;
	border-radius $radius;
}

.alert {
	@include border-radius(4px);
}
					

/* Output */
.alert {
	-webkit-border-radius: 4px;
	-moz-border-radius: 4px;
	border-radius 4px;
}
